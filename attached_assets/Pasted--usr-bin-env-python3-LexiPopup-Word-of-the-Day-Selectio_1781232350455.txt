# ============================================================
# SELECTION CRITERIA (Based on your requirements)
# ============================================================

class WordOfTheDaySelector:
    """
    Selects Word of the Day based on MULTIPLE criteria:
    1. Frequency: 30-70% (not too common, not too rare)
    2. Difficulty: Intermediate or Advanced (not Beginner, not Expert)
    3. Length: 5-12 letters (perfect size)
    4. Daily use potential: Medium to High
    5. Surprise/Value factor: Words people should know but don't
    """
    
    def __init__(self, words_db: List[Tuple]):
        self.words_db = words_db
    
    def calculate_word_score(self, word_data: Tuple) -> int:
        """
        Calculate a score for each word based on ALL criteria.
        Higher score = better Word of the Day candidate.
        """
        word, meaning, frequency, difficulty, length, daily_use = word_data
        
        score = 0
        
        # CRITERION 1: Frequency (30-70% is ideal)
        # Perfect score at 45% frequency
        if 30 <= frequency <= 70:
            # Closer to 45% = higher score
            freq_score = 100 - abs(frequency - 45) * 2
            score += freq_score
        else:
            score += 10  # Very low score if outside range
        
        # CRITERION 2: Difficulty (2=Intermediate, 3=Advanced)
        if difficulty in [2, 3]:
            score += 100 if difficulty == 2 else 80
        else:
            score += 20  # Beginner (1) or Expert (4) not preferred
        
        # CRITERION 3: Length (5-12 letters is ideal)
        if 5 <= length <= 12:
            # Perfect length around 7-9 letters
            if 7 <= length <= 9:
                score += 100
            else:
                score += 70
        else:
            score += 20
        
        # CRITERION 4: Daily use potential (7+ is good)
        score += daily_use * 10
        
        # CRITERION 5: Bonus - Words ending with specific patterns
        # (belittle, ephemeral type patterns)
        if word.endswith("le") or word.endswith("al") or word.endswith("ty"):
            score += 15
        
        # CRITERION 6: Penalty for very long or very short words
        if length > 14 or length < 4:
            score -= 50
        
        return score
    
    def get_candidate_words(self) -> List[Tuple]:
        """Filter and score all words, return sorted candidates."""
        candidates = []
        
        for word_data in self.words_db:
            score = self.calculate_word_score(word_data)
            candidates.append((word_data, score))
        
        # Sort by score (highest first)
        candidates.sort(key=lambda x: x[1], reverse=True)
        return candidates
    
    def get_word_of_day(self, date: datetime.date) -> Dict:
        """
        Select Word of the Day based on date.
        Same date = same word for ALL users (Global mode).
        """
        # Get all candidates sorted by score
        candidates = self.get_candidate_words()
        
        # Use date as seed for deterministic selection
        # This ensures same word for everyone on same day
        day_of_year = date.timetuple().tm_yday  # 1 to 365
        random.seed(day_of_year)
        
        # Pick from top 20 candidates (the very best ones)
        # So word is always high quality, but varies by day
        top_candidates = candidates[:20]
        
        # Add some randomness based on date
        selected_index = day_of_year % len(top_candidates)
        selected_word_data, score = top_candidates[selected_index]
        
        word, meaning, frequency, difficulty, length, daily_use = selected_word_data
        
        # Map difficulty number to name
        difficulty_map = {1: "Beginner", 2: "Intermediate", 3: "Advanced", 4: "Expert"}
        
        # Map frequency to description
        if frequency < 30:
            freq_desc = "Rare"
        elif frequency < 50:
            freq_desc = "Less Common"
        elif frequency < 70:
            freq_desc = "Moderate"
        else:
            freq_desc = "Common"
        
        return {
            "word": word,
            "meaning": meaning,
            "frequency": frequency,
            "frequency_desc": freq_desc,
            "difficulty": difficulty_map[difficulty],
            "length": length,
            "daily_use_score": daily_use,
            "selection_score": score,
            "date": date.isoformat(),
            "reason": self._get_reason(frequency, difficulty, length)
        }
    
    def _get_reason(self, frequency: int, difficulty: int, length: int) -> str:
        """Generate human-readable reason why this word was chosen."""
        reasons = []
        
        if 30 <= frequency <= 70:
            reasons.append(f"frequency is {frequency}% (perfect - not too common, not too rare)")
        else:
            reasons.append(f"frequency is {frequency}%")
        
        if difficulty in [2, 3]:
            reasons.append("difficulty is perfect (Intermediate/Advanced level)")
        
        if 5 <= length <= 12:
            reasons.append(f"length is {length} letters (ideal for learning)")
        
        return " | ".join(reasons)

def print_word_of_day(word_info: Dict):

    print(f"   🌟  {word_info['word'].upper()}  🌟")
    print()
    print(f"   📖 Meaning: {word_info['meaning']}")
    print()
    print(f"   📊 Stats:")
    print(f"      • Frequency: {word_info['frequency']}% ({word_info['frequency_desc']})")
    print(f"      • Difficulty: {word_info['difficulty']}")
    print(f"      • Length: {word_info['length']} letters")
    print(f"      • Daily Use Score: {word_info['daily_use_score']}/10")
    print(f"      • Selection Score: {word_info['selection_score']}")
    print()
    print(f"   ✅ Why this word?: {word_info['reason']}")
    print()
    print("=" * 65)


def print_candidates_ranking(selector: WordOfTheDaySelector):
    """Show top candidates with their scores."""
    print("\n" + "=" * 65)
    print("🏆 TOP CANDIDATES FOR WORD OF THE DAY")
    print("(Based on: Frequency 30-70%, Difficulty 2-3, Length 5-12)")
    print("=" * 65)
    print()
    print(f"{'Rank':<6} {'Word':<15} {'Score':<8} {'Freq':<6} {'Diff':<12} {'Len':<6} {'Daily Use':<10}")
    print("-" * 65)
    
    candidates = selector.get_candidate_words()
    for i, (word_data, score) in enumerate(candidates[:15], 1):
        word, meaning, freq, diff, length, daily = word_data
        diff_name = {1: "Beginner", 2: "Intermediate", 3: "Advanced", 4: "Expert"}[diff]
        print(f"{i:<6} {word:<15} {score:<8} {freq:<6} {diff_name:<12} {length:<6} {daily}/10")
    
    print("\n" + "=" * 65)
    print("✅ Words with score > 200 are PERFECT Word of Day candidates!")
    print("   Examples: belittle, ephemeral, serendipity, tenacious")
    print("=" * 65)
