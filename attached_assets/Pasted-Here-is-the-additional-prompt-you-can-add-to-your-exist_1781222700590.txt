## 🔄 DRAG-TO-REORDER IMPLEMENTATION

The user can drag any layer to change its priority:

```kotlin
// ReorderableList implementation using Jetpack Compose
@Composable
fun LayerReorderList(
    layers: List<LookupLayer>,
    onOrderChanged: (List<LookupLayer>) -> Unit
) {
    Column {
        layers.forEach { layer ->
            ReorderableItem(
                onMove = { from, to ->
                    val newOrder = layers.toMutableList()
                    newOrder.add(to, newOrder.removeAt(from))
                    onOrderChanged(newOrder)
                }
            ) {
                LayerCard(layer)
            }
        }
    }
}
```

🚦 LOOKUP FLOW WITH CUSTOMIZATION

The lookup engine must respect user's layer order and enabled status:

```kotlin
class CustomizableLookupEngine(
    private val layerOrderConfig: LayerOrderConfig
) {
    suspend fun lookupWord(word: String): LookupResult {
        val activeLayers = layerOrderConfig.layerOrder
            .filter { layerOrderConfig.enabledLayers[it] == true }
        
        for (layerName in activeLayers) {
            val result = when (layerName) {
                "cache" -> cacheLayer.lookup(word)
                "offline_db" -> offlineDbLayer.lookup(word)
                "online_api" -> onlineApiLayer.lookup(word)
                "groq_ai" -> groqAILayer.lookup(word)
                "openai" -> openAILayer.lookup(word)
                "on_device_ai" -> onDeviceAILayer.lookup(word)
                "rule_based" -> ruleBasedLayer.lookup(word)
                else -> null
            }
            
            if (result != null) {
                // Cache result from higher priority layer
                cacheLayer.save(word, result)
                return result
            }
        }
        
        return LookupResult.NotFound(
            message = "Word not found. Enable more layers in Settings."
        )
    }
}
```
