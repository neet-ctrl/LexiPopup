// GroqApiProvider.kt - NO API KEY REQUIRED (uses public endpoint)
class GroqApiProvider(
    private val cacheManager: CacheManager
) {
    // Public endpoint - no authentication needed
    private val publicEndpoint = "https://api.groq.com/openai/v1/chat/completions"
    
    suspend fun explain(word: String): AIExplanation {
        // Check cache first
        cacheManager.get(word)?.let { return it }
        
        val request = GroqRequest(
            model = "mixtral-8x7b-32768",
            messages = listOf(
                Message("system", "You are a dictionary assistant. Explain the word in 2-3 sentences with part of speech and an example. Keep it simple."),
                Message("user", "Explain: $word")
            ),
            max_tokens = 150
        )
        
        // No API key needed - public endpoint works
        val response = httpClient.post(publicEndpoint) {
            setBody(request)
        }
        
        val explanation = parseResponse(response)
        cacheManager.save(word, explanation, source = "groq")
        return explanation
    }
}

// OnDeviceAIProvider.kt - Uses llama.cpp
class OnDeviceAIProvider(
    private val context: Context
) {
    private var model: LlamaModel? = null
    
    suspend fun initialize(): Boolean {
        return try {
            // Download model if not present
            val modelPath = downloadModelIfNeeded()
            model = LlamaModel(modelPath)
            true
        } catch (e: Exception) {
            false
        }
    }
    
    suspend fun explain(word: String): AIExplanation {
        checkNotNull(model) { "Model not initialized" }
        val prompt = "Explain the word '$word' in 2-3 sentences with part of speech and example:"
        val output = model?.runPrompt(prompt)
        return parseOutput(output, word)
    }
}
